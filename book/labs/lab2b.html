start_date = "2024-09-10T12:00:00"

data = "GOES-16"
scan = "full_disk"

timelapse = geemap.goes_timelapse(
    roi, "goes.gif", start_date, end_date, data, scan, framesPerSecond=5, overlay_data="us_states"
)
geemap.show_image(timelapse)
